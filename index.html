settings.breakTime = parseInt(document.getElementById('breakTime').value); settings.trackCycles = document.getElementById('trackCycles').checked; settings.cyclesBeforeLongBreak = parseInt(document.getElementById('cyclesBeforeLongBreak').value); settings.longBreak = parseInt(document.getElementById('longBreak').value); settings.enableNotifications = document.getElementById('enableNotifications').checked; // Reset timer with new settings timeLeft = (mode === 'focus' ? settings.focusTime : settings.breakTime) * 60;
