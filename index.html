progressRing.style.strokeDashoffset = offset; } // Format time as MM:SS function formatTime(seconds) { const minutes = Math.floor(seconds / 60); const remainingSeconds = seconds % 60; return `${minutes.toString().padStart(2, '0')}:${remainingSeconds.toString().padStart(2, '0')}`; } // Update timer display function updateDisplay() { timeDisplay.textContent = formatTime(timeLeft); const progress = ((totalTime - timeLeft) / totalTime) * 100; setProgress(progress); if (settings.trackCycles) { cycleDisplay.textContent = `Cycle ${cycleCount}/4`; }); } // Save settings function saveSettings() { const newSettings = { focusTime: parseInt(focusTimeInput.value), shortBreakTime: parseInt(shortBreakTimeInput.value), longBreakTime: parseInt(longBreakTimeInput.value), enableNotifications: notificationCheckbox.checked, trackCycles: trackCyclesCheckbox.checked, notificationIcon: settings.notificationIcon }; } catch (error) { console.error('Error sending test notification:', error); } return true; } else { notificationCheckbox.checked = false; settings.enableNotifications = false; if (permission === 'denied') { alert('Please enable notifications in your browser settings to receive Pomodoro alerts.'); settings.enableNotifications = false; return; } if (!await checkNotificationSupport()) { notificationSetting.classList.remove('visible'); return; } notificationSetting.classList.add('visible'); if (Notification.permission === 'granted') { notificationCheckbox.checked = settings.enableNotifications; }); saveSettingsButton.addEventListener('click', saveSettings); settingsModal.addEventListener('click', (e) => { if (e.target === settingsModal) { settingsModal.classList.remove('visible'); } }); document.getElementById('notificationPromptAllow').addEventListener('click', async () => { const permissionGranted = await requestNotificationPermission(); if (permissionGranted) { notificationCheckbox.checked = true; settings.enableNotifications = true; } }); document.getElementById('notificationPromptLater').addEventListener('click', () => { hideNotificationPrompt();
